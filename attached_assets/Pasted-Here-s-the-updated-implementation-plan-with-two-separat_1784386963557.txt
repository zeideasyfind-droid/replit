```python
from datetime import datetime, date
import re
import random

def normalize_availability(date_str):
    """Normalize availability date to 'Ready to occupy' if immediate/today/past"""
    if not date_str:
        return "Ready to occupy"
    
    date_str = str(date_str).lower().strip()
    
    # Check for immediate availability terms
    immediate_terms = ["immediate", "ready", "today", "now", "vacant", "available now"]
    if any(term in date_str for term in immediate_terms):
        return "Ready to occupy"
    
    # Check if date is in the past or today
    try:
        parsed_date = datetime.strptime(date_str, "%Y-%m-%d").date()
        if parsed_date <= date.today():
            return "Ready to occupy"
    except (ValueError, TypeError):
        pass
    
    return date_str


def normalize_tenant_preference(tenant_str, diet_pref=None):
    """Normalize tenant preference and add diet information"""
    tenant_str = str(tenant_str).lower() if tenant_str else ""
    
    # Normalize tenant types
    has_family = "family" in tenant_str
    has_professional = any(term in tenant_str for term in ["working", "professional"])
    has_anyone = any(term in tenant_str for term in ["anyone", "any", "open"])
    
    # Build normalized tenant string
    if has_anyone or (not has_family and not has_professional):
        result = "Anyone"
    else:
        parts = []
        if has_family:
            parts.append("Families")
        if has_professional:
            parts.append("Working Professionals")
        result = " / ".join(parts) if parts else "Anyone"
    
    # Add diet preference
    if diet_pref:
        diet_pref = str(diet_pref).lower().strip()
        if "vegetarian" in diet_pref or "veg" in diet_pref:
            result += ", Vegetarian"
        elif "non-vegetarian" in diet_pref or "non-veg" in diet_pref:
            result += ", Non-Vegetarian"
    
    return result


def format_title_with_utility(title, property_details):
    """Add 'with Utility' to title if fully furnished 2 BHK with 2 bathrooms & balcony"""
    furnishing = property_details.get('furnishing', '').lower()
    
    if (furnishing == 'fully furnished' and 
        '2 bhk' in title.lower() and 
        '2 bathroom' in title.lower() and 
        'balcony' in title.lower()):
        return title + " with Utility"
    
    return title


def generate_drive_folder_name(property_details):
    """
    Generate Drive folder name following EasyFind format:
    Format: EFF-{numeric}-{society/apt OR location-owner}
    """
    society = property_details.get('society', '').strip()
    apartment = property_details.get('apartment', '').strip()
    location = property_details.get('location', '').strip()
    owner = property_details.get('owner', 'Unknown').strip()
    property_name = property_details.get('property_name', '').strip()
    community_type = property_details.get('community_type', 'Gated').lower()
    
    # Generate numeric identifier (random 3-digit)
    numeric_id = str(random.randint(100, 999))
    
    # Determine base name
    if society or apartment:
        # Society/apartment: use society or apartment name
        base_name = society if society else apartment
        folder_name = f"EFF-{numeric_id}-{base_name}"
    elif property_name:
        # Property name exists: property name - owner
        folder_name = f"EFF-{numeric_id}-{property_name}-{owner}"
    elif community_type in ['semi-gated', 'non-society', 'non-gated']:
        # Semi-gated/non-society: location - owner
        folder_name = f"EFF-{numeric_id}-{location}-{owner}"
    else:
        # Default: location - owner
        folder_name = f"EFF-{numeric_id}-{location}-{owner}"
    
    # Clean up folder name (remove special characters)
    folder_name = re.sub(r'[^a-zA-Z0-9\s\-]', '', folder_name)
    folder_name = re.sub(r'\s+', ' ', folder_name).strip()
    
    return folder_name


def format_whatsapp_message(property_details):
    """
    Format a single property into WhatsApp message format
    """
    # Apply all normalizations
    title = format_title_with_utility(
        property_details.get('title', ''),
        property_details
    )
    
    available_from = normalize_availability(property_details.get('available_from', ''))
    preferred_tenant = normalize_tenant_preference(
        property_details.get('preferred_tenant', 'Anyone'),
        property_details.get('diet_preference', '')
    )
    
    # Build message
    message = f"""*{title}*

Rent: ₹{property_details.get('rent', 'N/A')}
Maintenance: ₹{property_details.get('maintenance', 'N/A')}
Deposit: ₹{property_details.get('deposit', 'N/A')}
Area: {property_details.get('sqft', 'N/A')} Sq. Ft.
Floor: {property_details.get('floor', 'N/A')}
Available From: {available_from}
Preferred Tenant: {preferred_tenant}
Pets: {property_details.get('pets', 'Not Allowed')}
Community: {property_details.get('community', 'Gated')}
Location: *{property_details.get('location', 'N/A')}*

📸 *Gallery:* {property_details.get('gallery_link', 'N/A')}

*{property_details.get('property_name', property_details.get('society', 'N/A'))}*
📍 Map: {property_details.get('map_link', 'N/A')}"""
    
    return message
```

```python
from utils.formatters import format_whatsapp_message, generate_drive_folder_name
import json
from datetime import datetime

@app.route('/single_upload', methods=['GET'])
def single_upload_page():
    """Render single property upload page"""
    return render_template('single_upload.html')

@app.route('/api/preview-single', methods=['POST'])
def preview_single():
    """Generate preview for single property"""
    try:
        property_data = request.get_json()
        
        # Format message
        message = format_whatsapp_message(property_data)
        
        # Generate Drive folder name
        drive_folder_name = generate_drive_folder_name(property_data)
        
        return jsonify({
            'success': True,
            'message': message,
            'drive_folder_name': drive_folder_name
        })
        
    except Exception as e:
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500

@app.route('/api/generate-single', methods=['POST'])
def generate_single():
    """Generate and save single property files"""
    try:
        property_data = request.get_json()
        
        # Format message
        message = format_whatsapp_message(property_data)
        
        # Generate Drive folder name
        drive_folder_name = generate_drive_folder_name(property_data)
        
        # Create timestamp for unique filenames
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        
        # Save WhatsApp message
        whatsapp_filename = f'output/whatsapp_{timestamp}.txt'
        with open(whatsapp_filename, 'w') as f:
            f.write(message)
        
        # Save property data as JSON
        json_filename = f'output/property_{timestamp}.json'
        with open(json_filename, 'w') as f:
            json.dump({
                'property_data': property_data,
                'drive_folder_name': drive_folder_name,
                'generated_message': message,
                'timestamp': timestamp
            }, f, indent=2)
        
        return jsonify({
            'success': True,
            'whatsapp_file': whatsapp_filename,
            'json_file': json_filename,
            'drive_folder_name': drive_folder_name
        })
        
    except Exception as e:
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500
```

```python
from services.bulk_processor import BulkPropertyProcessor
import zipfile
import os

@app.route('/bulk_upload', methods=['GET'])
def bulk_upload_page():
    """Render bulk property upload page"""
    return render_template('bulk_upload.html')

@app.route('/api/preview-bulk', methods=['POST'])
def preview_bulk():
    """Generate preview for all bulk properties"""
    try:
        data = request.get_json()
        properties = data.get('properties', [])
        
        previews = []
        for prop in properties:
            message = format_whatsapp_message(prop)
            drive_folder = generate_drive_folder_name(prop)
            previews.append({
                'title': prop.get('title'),
                'message': message,
                'drive_folder_name': drive_folder
            })
        
        return jsonify({
            'success': True,
            'previews': previews
        })
        
    except Exception as e:
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500

@app.route('/api/generate-bulk', methods=['POST'])
def generate_bulk():
    """Generate and save all bulk properties"""
    try:
        data = request.get_json()
        properties = data.get('properties', [])
        
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        
        # Initialize bulk processor
        processor = BulkPropertyProcessor()
        results = processor.process_properties_batch(properties)
        
        # Create ZIP file with all outputs
        zip_filename = f'output/bulk_{timestamp}.zip'
        with zipfile.ZipFile(zip_filename, 'w') as zipf:
            # Add individual property files
            for idx, msg_data in enumerate(results['messages']):
                # WhatsApp message
                msg_file = f'property_{idx+1}_whatsapp.txt'
                zipf.writestr(msg_file, msg_data['message'])
                
                # JSON data
                json_file = f'property_{idx+1}_data.json'
                zipf.writestr(json_file, json.dumps({
                    'property': properties[idx],
                    'drive_folder': results['drive_folders'][idx]['folder_name'] if idx < len(results['drive_folders']) else None
                }, indent=2))
        
        # Save combined JSON
        combined_json = f'output/bulk_data_{timestamp}.json'
        with open(combined_json, 'w') as f:
            json.dump(results, f, indent=2)
        
        # Save combined TXT
        combined_txt = f'output/bulk_messages_{timestamp}.txt'
        with open(combined_txt, 'w') as f:
            for msg_data in results['messages']:
                f.write(f"=== {msg_data['title']} ===\n")
                f.write(f"{msg_data['message']}\n\n")
                f.write("-" * 80 + "\n\n")
        
        return jsonify({
            'success': True,
            'processed_count': results['processed_count'],
            'total_properties': len(properties),
            'zip_file': zip_filename,
            'json_file': combined_json,
            'txt_file': combined_txt,
            'errors': results['errors']
        })
        
    except Exception as e:
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500
```
